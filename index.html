    <script>

        document.getElementById("screeningForm").addEventListener("submit", function(event) {

            event.preventDefault();

            let gender = document.getElementById("gender").value;

            let smoking = document.getElementById("smoking_status").value;

                screeningMessage += "<li>Colorectal Cancer Screening</li>";

            }

                screeningMessage += "<li>Mammogram for Breast Cancer</li>";

            }

            if (smoking === "Smoker") {

                screeningMessage += "<li>Lung Cancer Screening</li>";

            }

            if (bmi >= 30) {

                screeningMessage += "<li>Diabetes and Heart Disease Screening</li>";

            }

            

            screeningMessage += "</ul>";


            let resultDiv = document.getElementById("result");

            resultDiv.innerHTML = screeningMessage;

            resultDiv.style.display = "block";

        });
